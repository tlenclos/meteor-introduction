Templates

{{> forecast}}
<template name="forecast">
<div>It'll be {{prediction}} tonight</div>
</template>


// in client/myapp.js: reactive helper function
Template.forecast.prediction = function () {
  return Session.get("weather");
};

Collections

// Create a new collection
Messages = new Meteor.Collection("messages");
// return array of my messages
var myMessages = Messages.find({userId: Session.get('myUserId')}).fetch();

// mark my first message as "important"
Messages.update(myMessages[0]._id, {$set: {important: true}});

// create a new message
Messages.insert({text: "Hello, world!"});

Comment ca marche?

  • Chaque pixel est stocké dans une collection
  • Synchronisation et rafraichissement du canvas

Contexte réactif

Deps.autorun(function () {
    var pixels = PixelBoardsCollection.find({});
    _.each(pixels.fetch(), function(item) {
        if(!self.grid[item.y])
            self.grid[item.y] = [];

        self.grid[item.y][item.x] = item.color;
    });

    self.draw();
});
